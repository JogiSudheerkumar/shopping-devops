pipeline {
    agent any

    stages {
        stage('Build Docker Image') {
            steps {
                sh 'docker build -t shopping-app .'
            }
        }

        stage('Stop Old Container') {
            steps {
                sh 'docker stop shopping_container || true'
                sh 'docker rm shopping_container || true'
            }
        }

        stage('Run New Container') {
            steps {
                sh 'docker run -d -p 5000:5000 --name shopping_container shopping-app'
            }
        }
    }
}
